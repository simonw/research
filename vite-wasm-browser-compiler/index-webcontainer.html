Vite Single File Bundler WebContainer

Run real Vite with vite-plugin-singlefile entirely in your browser using StackBlitz WebContainers.

How it works: This tool boots a full Node.js environment in your browser using StackBlitz WebContainers. It then runs Vite with vite-plugin-singlefile to bundle your page into a single HTML file.
Cross-Origin Isolation Required: This page requires special headers to enable WebContainers. Please use the server-coi.js server.

        function copyOutput() {
            navigator.clipboard.writeText(document.getElementById('outputArea').value).then(() => {
                setStatus('Copied to clipboard!', 'success');
                log('Output copied to clipboard', 'success');
            });
        }

        function downloadOutput() {
            const blob = new Blob([bundledOutput], { type: 'text/html' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'bundled-page.html';
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            log('Downloaded bundled HTML', 'success');
        }

        function useSampleProject() {
            document.getElementById('urlInput').value = '(Sample Vite Project)';
            document.getElementById('urlInput').dataset.useSample = 'true';
            log('Sample project selected - click Bundle to build', 'info');
        }
